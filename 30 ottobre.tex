\section{Nuovi operatori}
\subsection*{Approfondimento sull'operatore momento angolare} 

\begin{enumerate}
    \item $[J_x, J_y] = i\hbar J_z$
    \item $[J_y, J_z] = i\hbar J_x$
    \item $[J_z, J_x] = i\hbar J_y$
    \item $[J^2, J_x] = [J^2, J_y] = [J^2, J_z] = 0$
    \item $[J_z, J_{\pm}] = \pm\hbar J_{\pm}$
    \item $[J_+, J_-] = 2\hbar J_z$
    \item $J_{\pm} = J_x \pm iJ_y$
    \item $[J^2, J_{\pm}] = 0$
\end{enumerate}
